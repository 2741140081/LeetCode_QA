-- =====================================================
-- 小偷游戏数据库表结构
-- 创建日期：2026-03-24
-- 描述：用于存储小偷游戏存档、BOSS 映射、挑战记录和难度配置
-- =====================================================

-- 1. 小偷游戏存档建筑物表
-- 存储每个游戏的存档建筑物信息，包括每日挑战上限
CREATE TABLE IF NOT EXISTS `thief_game_archiver` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '自增主键 ID',
    `game_name` VARCHAR(100) NOT NULL COMMENT '游戏名称',
    `archiver_number` INT NOT NULL COMMENT '存档编号 (1-6)',
    `archiver_name` VARCHAR(100) NOT NULL COMMENT '存档建筑物名称',
    `archiver_folder` VARCHAR(255) NOT NULL COMMENT '存档建筑物文件夹路径',
    `daily_challenge_limit` INT DEFAULT 0 COMMENT '每日挑战上限次数',
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `updated_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    UNIQUE KEY `uk_game_archiver` (`game_name`, `archiver_number`),
    INDEX `idx_game_name` (`game_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='小偷游戏存档建筑物表';

-- 2. 小偷游戏存档 BOSS 映射表
-- 存储存档建筑物与存档 BOSS 的映射关系及图片路径
CREATE TABLE IF NOT EXISTS `thief_game_archiver_boss` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '自增主键 ID',
    `game_name` VARCHAR(100) NOT NULL COMMENT '游戏名称',
    `archiver_number` INT NOT NULL COMMENT '存档建筑编号 (1-6)',
    `archiver_name` VARCHAR(100) NOT NULL COMMENT '存档建筑名称',
    `boss_name` VARCHAR(100) NOT NULL COMMENT '存档 BOSS 名称',
    `image_path` VARCHAR(500) NOT NULL COMMENT 'BOSS 图片所在路径',
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `updated_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX `idx_game_archiver` (`game_name`, `archiver_number`),
    INDEX `idx_archiver_name` (`archiver_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='小偷游戏存档 BOSS 映射表';

-- 3. 小偷游戏存档 BOSS 挑战记录表
-- 记录每次挑战存档 BOSS 的结果，用于统计每日挑战次数
CREATE TABLE IF NOT EXISTS `thief_game_challenge_record` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '自增主键 ID',
    `game_name` VARCHAR(100) NOT NULL COMMENT '游戏名称',
    `archiver_number` INT NOT NULL COMMENT '存档建筑物编号 (1-6)',
    `challenge_result` TINYINT NOT NULL DEFAULT 0 COMMENT '挑战结果：0-失败，1-成功',
    `challenge_time` DATETIME NOT NULL COMMENT '挑战完成时间',
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    INDEX `idx_game_archiver_time` (`game_name`, `archiver_number`, `challenge_time`),
    INDEX `idx_challenge_date` (`challenge_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='小偷游戏存档 BOSS 挑战记录表';

-- 4. 小偷游戏难度配置表
-- 存储每个难度的最大可执行次数和额外吞噬装备次数
CREATE TABLE IF NOT EXISTS `thief_game_difficulty_config` (
    `id` BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '自增主键 ID',
    `difficulty_level` INT NOT NULL COMMENT '难度等级 (1-50)',
    `game_name` VARCHAR(100) NOT NULL COMMENT '游戏名称',
    `difficulty_image_path` VARCHAR(500) DEFAULT NULL COMMENT '难度图片地址',
    `max_execution_count` INT NOT NULL DEFAULT 0 COMMENT '最大可执行次数',
    `current_execution_count` INT DEFAULT 0 COMMENT '已执行次数',
    `extra_devour_times` INT DEFAULT 0 COMMENT '额外需要进行吞噬装备次数 (默认 0)',
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `updated_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    UNIQUE KEY `uk_game_difficulty` (`game_name`, `difficulty_level`),
    INDEX `idx_difficulty_level` (`difficulty_level`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='小偷游戏难度配置表';

-- 插入小偷游戏的难度配置示例数据 (难度 1-50)
INSERT INTO `thief_game_difficulty_config` (`difficulty_level`, `game_name`, `difficulty_image_path`, `max_execution_count`, `extra_devour_times`)
SELECT
    difficulty_level,
    '小偷偷金' AS game_name,
    CONCAT('difficulty_', difficulty_level, '.png') AS difficulty_image_path,
    difficulty_level * 1 AS max_execution_count,  -- 难度 N 对应 N 次上限
    0 AS extra_devour_times
FROM (
    SELECT @row := @row + 1 AS difficulty_level
    FROM information_schema.COLUMNS, (SELECT @row := 0) r
    WHERE @row < 50
) AS levels
ON DUPLICATE KEY UPDATE
    `max_execution_count` = VALUES(`max_execution_count`),
    `extra_devour_times` = VALUES(`extra_devour_times`);
